# 枚举与不可枚举

# 1. 什么是枚举

枚举是指对象中的属性是否可以被遍历出来。


# 2. 判断属性是否可枚举

在 JavaScript 中，对象的属性分为可枚举和不可枚举之分，它们是由属性的 enumerable 值决定的。

JS 中的基本包装类型的原型属性是不可枚举的，比如 Object、Array、Number 等。

let num = new Number()

for(let pro in num) {
   console.log('num.' + pro + '=' + num[pro])
}

它的输出结果是空的，因为 Number 中的内置属性是不可枚举的，所以不能被 for ... in 访问到。


propertyIsEnumerable('属性名') 方法可以用来判断属性是否可枚举。

如果是对象原型中的属性，不管是否可枚举，propertyIsEnumerable() 都会返回 false，但是 for...in 仍然可以读出原型链中的可枚举属性。

function Person() {
  this.name = '我是实例属性'
  this.age = 19
}

let p = new Person()
console.log(p.propertyIsEnumerable('name')) // true
 
// 添加一个原型属性
Person.prototype.prop = '我是原型属性'
console.log(p.propertyIsEnumerable('prop')) // prop 是原型上的属性，所以返回的是 false
 
for(let k in p){
  console.log(k + ',' + p[k]) // name,我是实例属性  age,19  prop,我是原型属性
}

# 3. 枚举方法的区别

  • for…in

    可以遍历出对象本身具有的属性、通过 Object.defineProperty() 方法加的可枚举属性或通过原型对象绑定的可以枚举属性。

    function enumer() {
  this.a = '我是对象本身就有的属性'
}
enumer.prototype.b = '我是通过原型对象挂载的属性'

let fn = new enumer()
Object.defineProperty(fn, 'c', {
  value: '我是通过Object.defineProperty方法添加的可枚举属性',
  enumerable: true  // 是否可以枚举
})

for(let item in fn) {
  console.log(item)
}
/* 输出 */
// a
// c
// b

  • Object.keys()

    可以枚举对象本身的属性和通过 Object.defineProperty() 添加的可枚举属性。

    function enumer() {
  this.a = '我是对象本身就有的属性'
}

enumer.prototype.b = '我是通过原型对象挂载的属性'

let fn = new enumer()
Object.defineProperty(fn, 'c', {
  value: '我是通过Object.defineProperty方法添加的可枚举属性',
  enumerable: true
})

console.log(Object.keys(fn))
/* 输出 */
// ['a', 'c']

  • JSON.stringify()

    只能序列化本身的属性和通过 Object.defineProperty() 添加的可枚举属性为 JSON 对象。

    function enumer() {
  this.a = '我是对象本身就有的属性'
}

enumer.prototype.b = '我是通过原型对象挂载的属性'

let fn = new enumer()
Object.defineProperty(fn, 'c', {
  value: '我是通过Object.defineProperty方法添加的可枚举属性',
  enumerable: true
})

console.log(JSON.stringify(fn))
/* 输出 */
// {"a":"我是对象本身就有的属性","c":"我是通过Object.defineProperty方法添加的可枚举属性"}
