# Token 无感刷新

# 1. 使用场景

用户正在填写一个内容很多的表单，费时费力的填写完成后点击提交，此时刚好 token 过期，如果直接跳转到登录页，那就会丢失用户填写的内容，体验非常不好。

所以要实现用户在操作过程中，无感刷新过期 token，继续执行用户的请求，确保不会因 token 过期跳转登录页而造成数据丢失。


# 2. 实现思路

借助 axios 的请求拦截器和响应拦截器进行实现。

请求拦截器中：

  1. 设置一个活跃时长，token 过期后，如果在这个活跃时长内还有请求，就去刷新 token，如果超过这个时长再去请求，则执行退出登录操作。

  2. 为了防止多个请求同时执行刷新 token 操作，需要设置一个标识来判断当前是否正在刷新 token。

  3. 一个页面一般会有很多请求，当其中某一个请求开始执行刷新 token 操作时，其他请求暂时挂起，放到一个缓存队列中，等 token 刷新完成后再依次执行缓存队列中的请求。


响应拦截器中：

如果接口返回 401 或 403，则执行退出登录操作。


# 3. 具体代码

import axios from 'axios'
import { parseToken } from '@/utils/auth' // 解析 token 的方法

// 请求头不需要携带 token 的请求
const uapi = axios.create({ 
  timeout: 60000
})

// 请求头需要携带 token 的请求
const api = axios.create({ 
  timeout: 60000
})

let isRefreshing = false // 是否正在刷新 token 的标识
let cacheRequests = [] // 缓存请求队列

// 请求拦截器
api.interceptors.request.use(
  config => {
    if (localStorage.getItem('AccessToken')) {
      const accessExp = parseToken(localStorage.getItem('AccessToken')).exp // AccessToken 过期时间
      const activePeriod = 7200 // 活跃时长：2h
      const currentTime = Math.round(new Date().getTime() / 1000) // 转换成 unix 时间戳
      if (currentTime > accessExp) {
        if (currentTime - accessExp < activePeriod) { // 活跃时长内刷新 token
          if (!isRefreshing) {
            isRefreshing = true
            uapi.post('/refresh_token', { // 执行刷新 token 接口
              accessToken: localStorage.getItem('AccessToken'),
              refreshToken: localStorage.getItem('RefreshToken')
            }).then(res => {
              localStorage.setItem('AccessToken', res.data.data.accessToken)
              localStorage.setItem('RefreshToken', res.data.data.refreshToken)
              config.headers.Authentication = localStorage.getItem('AccessToken') // 请求头中替换新的 token
              cacheRequests.forEach(cb => cb(res.data.data.accessToken)) // 依次执行挂起的请求
              cacheRequests = [] // 请求完后清空缓存请求队列
            }).catch(() => {
              // 执行退出登录操作
              // ...
            }).finally(() => {
              isRefreshing = false
            })
            // 此处一定要将第一个请求刷新 token 的接口放到缓存队列中，不然没法继续此请求
            return new Promise(resolve => {
              cacheRequests.push(token => {
                config.headers.Authentication = token //  请求头中替换新的 token
                resolve(config)
              })
            })
          } else {
            // 将同时请求的接口放入缓存队列
            return new Promise(resolve => {
              cacheRequests.push(token => {
                config.headers.Authentication = token
                // 将请求挂起
                resolve(config)
              })
            })
          }
        } else {
          // 执行退出登录操作
          // ...
        }
      } else {
        config.headers.Authentication = localStorage.getItem('AccessToken')
        return config
      }
    } else {
      return config
    }
  },
  error => {
    return Promise.reject(error)
  }
)

// 响应拦截器
api.interceptors.response.use(
  response => {
    return response
  },
  error => {
    if (error.response.data.code === 401 || error.response.data.code === 403) {
      // 执行退出登录操作
      // ...
    }
  }
)
